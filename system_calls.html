System Calls Of Windows

1. CreateFile:
- Description: Creates, opens, or truncates a file or device and returns a handle that can be used to access the file or device.
- Syntax:

HANDLE CreateFile(
  LPCTSTR lpFileName,
  DWORD dwDesiredAccess,
  DWORD dwShareMode,
  LPSECURITY_ATTRIBUTES lpSecurityAttributes,
  DWORD dwCreationDisposition,
  DWORD dwFlagsAndAttributes,
  HANDLE hTemplateFile
);

2. ReadFile:
- Description: Reads data from the specified file or device into a buffer.
- Syntax:

BOOL ReadFile(
  HANDLE hFile,
  LPVOID lpBuffer,
  DWORD nNumberOfBytesToRead,
  LPDWORD lpNumberOfBytesRead,
  LPOVERLAPPED lpOverlapped
);

3. WriteFile:
- Description: Writes data from a buffer to the specified file or device.
- Syntax:

BOOL WriteFile(
  HANDLE hFile,
  LPCVOID lpBuffer,
  DWORD nNumberOfBytesToWrite,
  LPDWORD lpNumberOfBytesWritten,
  LPOVERLAPPED lpOverlapped
);

4. CloseHandle:
- Description: Closes an open object handle.
- Syntax:

BOOL CloseHandle(
  HANDLE hObject
);

5. CreateProcess:
- Description: Creates a new process and its primary thread.
- Syntax:

BOOL CreateProcess(
  LPCTSTR lpApplicationName,
  LPTSTR lpCommandLine,
  LPSECURITY_ATTRIBUTES lpProcessAttributes,
  LPSECURITY_ATTRIBUTES lpThreadAttributes,
  BOOL bInheritHandles,
  DWORD dwCreationFlags,
  LPVOID lpEnvironment,
  LPCTSTR lpCurrentDirectory,
  LPSTARTUPINFO lpStartupInfo,
  LPPROCESS_INFORMATION lpProcessInformation
);

6. CreateThread:
- Description: Creates a new thread for execution within the calling process.
- Syntax:

HANDLE CreateThread(
  LPSECURITY_ATTRIBUTES lpThreadAttributes,
  SIZE_T dwStackSize,
  LPTHREAD_START_ROUTINE lpStartAddress,
  LPVOID lpParameter,
  DWORD dwCreationFlags,
  LPDWORD lpThreadId
);

7. VirtualAlloc:
- Description: Reserves, commits, or reserves and commits a region of memory within the virtual address space of a specified process.
- Syntax:

LPVOID VirtualAlloc(
  LPVOID lpAddress,
  SIZE_T dwSize,
  DWORD flAllocationType,
  DWORD flProtect
);

8. VirtualFree:
- Description: Releases, decommits, or releases and decommits a region of memory within the virtual address space of a specified process.
- Syntax:

BOOL VirtualFree(
  LPVOID lpAddress,
  SIZE_T dwSize,
  DWORD dwFreeType
);

9. CreateMutex:
- Description: Creates or opens a named or unnamed mutex object.
- Syntax:

HANDLE CreateMutex(
  LPSECURITY_ATTRIBUTES lpMutexAttributes,
  BOOL bInitialOwner,
  LPCTSTR lpName
);

10. WaitForSingleObject:
- Description: Waits until the specified object is in the signaled state or the time-out interval elapses.
- Syntax:

DWORD WaitForSingleObject(
  HANDLE hHandle,
  DWORD dwMilliseconds
);

11. ReleaseMutex:
- Description: Releases ownership of the specified mutex object.
- Syntax:

BOOL ReleaseMutex(
  HANDLE hMutex
);

12. RegOpenKeyEx:
- Description: Opens the specified registry key for reading or writing.
- Syntax:

LONG RegOpenKeyEx(
  HKEY hKey,
  LPCTSTR lpSubKey,
  DWORD ulOptions,
  REGSAM samDesired,
  PHKEY phkResult
);

13. RegQueryValueEx:
- Description: Retrieves the type and data for a specified value name associated with an open registry key.
- Syntax:

LONG RegQueryValueEx(
  HKEY hKey,
  LPCTSTR lpValueName,
  LPDWORD lpReserved,
  LPDWORD lpType,
  LPBYTE lpData,
  LPDWORD lpcbData
);

14. RegSetValueEx:
- Description: Sets the data and type of a specified value under a registry key.
- Syntax:

LONG RegSetValueEx(
  HKEY hKey,
  LPCTSTR lpValueName,
  DWORD Reserved,
  DWORD dwType,
  const BYTE lpData,
  DWORD cbData
);

15. ConnectNamedPipe:
- Description: Waits for a client to connect to a named pipe.
- Syntax:

BOOL ConnectNamedPipe(
  HANDLE hNamedPipe,
  LPOVERLAPPED lpOverlapped
);

16. DisconnectNamedPipe:
- Description: Disconnects the server end of a named pipe instance from a client process.
- Syntax:

BOOL DisconnectNamedPipe(
  HANDLE hNamedPipe
);

17. SendMessage:
- Description: Sends the specified message to a window or windows.
- Syntax:

LRESULT SendMessage(
  HWND hWnd,
  UINT Msg,
  WPARAM wParam,
  LPARAM lParam
);

18. PostMessage:
- Description: Places a message in the message queue associated with the thread that created the specified window.
- Syntax:

BOOL PostMessage(
  HWND hWnd,
  UINT Msg,
  WPARAM wParam,
  LPARAM lParam
);

19. GetTickCount:
- Description: Retrieves the number of milliseconds that have elapsed since the system was started.
- Syntax:

DWORD GetTickCount();

20. Sleep:
- Description: Suspends the execution of the current thread for a specified interval.
- Syntax:

void Sleep(
  DWORD dwMilliseconds
);
